\begin{tikzpicture}[node distance=2.5cm, on grid, auto]

  % Interpretation:
  % Each state q_r represents the remainder r (mod 4)
  % of the sum of input symbols read so far.
  % Alphabet: {0,1,2,3}

  % DFA states
  \node[state, initial] (q0) {$q_0$};                 % start: sum ≡ 0 (mod 4)
  \node[state, right of=q0] (q1) {$q_1$};
  \node[state, right of=q1] (q2) {$q_2$};
  \node[state, below of=q1, accepting] (q3) {$q_3$};  % accept: sum ≡ 3 (mod 4)

  % Transition rule:
  % From q_r on input d ∈ {0,1,2,3}, go to q_{(r + d) mod 4}.
  \path[->]
    (q0) edge[loop above] node {0} ()
         edge[above]      node {1} (q1)
         edge[bend left]  node {2} (q2)
         edge[bend left]  node {3} (q3)

    (q1) edge[loop above] node {0} ()
         edge[above]      node {1} (q2)
         edge[bend left]  node {2} (q3)
         edge[bend left]  node {3} (q0) % 1 + 3 ≡ 0 (mod 4)

    (q2) edge[loop above] node {0} ()
         edge[below]      node {1} (q3)
         edge[bend left]  node {2} (q0) % 2 + 2 ≡ 0 (mod 4)
         edge[bend left]  node {3} (q1)

    (q3) edge[loop below] node {0} ()
         edge[bend left]  node {1} (q0)
         edge[bend left]  node {2} (q1)
         edge[below]      node {3} (q2);
\end{tikzpicture}
